// Pins with the LED's
const int leds[] = {3 ,6 ,9};
const int ledCount = 3;

void setup() {
    for (int i = 0; i < ledCount; i++) {
    // Pins for LED's become Power
    pinMode (leds[i], OUTPUT);
    }
randomSeed(analogRead(0));
}

void loop () {
    int ledIndex = random(0, ledCount);
    int ledPin = leds[ledIndex];

    // brightness is between power level 55 and 255
    int brightness = random(55, 255);

    // onTime is random (between 0,1 and 0,5 seconds)
    int onTime = random(100, 500);
    // offTime is random(between o,1 and 0,4 seconds)
    int offTime = random(100, 400);

    // LED on
    analogWrite(ledPin, brightness);
    // delay tells the onTime
    delay(onTime);

    // LED off
    analogWrite(ledPin, LOW);
    // delay tells the offTime
    delay(offTime);
}
